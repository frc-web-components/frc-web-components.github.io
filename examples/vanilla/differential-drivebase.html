Examples for Differential Drivebase

With attributes


        <!-- Code for "With attributes" example -->
        <frc-differential-drivebase 
          left-motor-speed=".7"
          right-motor-speed="-.1"
        ></frc-differential-drivebase>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-number 
          key="/LiveWindow/Ungrouped/DifferentialDrive[1]/Left Motor Speed" 
          value="-.5"
        ></nt-number>

        <nt-number 
          key="/LiveWindow/Ungrouped/DifferentialDrive[1]/Right Motor Speed" 
          value=".5"
        ></nt-number>

        <frc-differential-drivebase 
          source-key="/LiveWindow/Ungrouped/DifferentialDrive[1]"
        ></frc-differential-drivebase>
