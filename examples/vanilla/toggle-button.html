With NetworkTables

/bool1 /button1 

        <!-- Code for "With NetworkTables" example -->

        <nt-boolean key="/button1/toggled" value="true"></nt-boolean>

        <frc-toggle-button source-key="/bool1">/bool1</frc-toggle-button>
        <frc-toggle-button source-key="/button1">/button1</frc-toggle-button>

With styles

Styled Button Styled Button

          .with-styles {

            color: white;

          }

          .with-styles[toggled]::part(button) {
            background: red;
            color: black;
          }
        </style>

        <frc-toggle-button class="with-styles">Styled Button</frc-toggle-button>
        <frc-toggle-button class="with-styles" toggled>Styled Button</frc-toggle-button>
