Examples for Voltage View

With attributes


        <!-- Code for "With attributes" example -->
        <frc-voltage-view value=".5"></frc-voltage-view>

        <frc-voltage-view value=".9" min="0"></frc-voltage-view>

        <frc-voltage-view 
          value="2"
          min="0"
          max="10"
          center="0"
          precision="4"
          num-tick-marks="3"
          unit="V"
        ></frc-voltage-view>

        <frc-voltage-view 
          value=".5"
          hide-text
          num-tick-marks="10"
        ></frc-voltage-view>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-number key="/voltView1/value" value="7"></nt-number>
        <nt-number key="/voltView1/min" value="0"></nt-number>
        <nt-number key="/voltView1/max" value="10"></nt-number>
        <nt-number key="/voltView1/precision" value="8"></nt-number>
        <nt-number key="/voltView1/center" value="0"></nt-number>
        <nt-boolean key="/voltView1/hideText" value="false"></nt-boolean>
        <nt-number key="/voltView1/numTickMarks" value="7"></nt-number>
        <nt-string key="/voltView1/unit" value="v"></nt-string>


        <frc-voltage-view source-key="/voltView1"></frc-voltage-view>

        <nt-number key="/num1" value=".8"></nt-number>

        <frc-voltage-view source-key="/num1"></frc-voltage-view>

With attributes and NetworkTables


        <!-- Code for "With attributes and NetworkTables" example -->
        <nt-number key="/num2" value="3"></nt-number>

        <frc-voltage-view source-key="/num2" max="3.5" hide-text></frc-voltage-view>

        <nt-number key="/voltView2/min" value="-20"></nt-number>
        <nt-number key="/voltView2/max" value="20"></nt-number>

        <frc-voltage-view 
          source-key="/voltView2"
          value="6" 
          num-tick-marks="2"
        ></frc-voltage-view>
