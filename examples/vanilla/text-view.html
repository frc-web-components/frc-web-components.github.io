With NetworkTables


        <!-- Code for "With NetworkTables" example -->

        <nt-number key="/value2" value="5"></nt-number>
        <nt-boolean key="/value3" value="false"></nt-boolean>
        <nt-string-array key="/value4" value='["string", "array"]'></nt-string-array>

        <nt-boolean-array key="/value6" value='[false,true]'></nt-boolean-array>
