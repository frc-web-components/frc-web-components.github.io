Examples for Number Slider

With attributes


        <!-- Code for "With attributes" example -->
        <frc-number-slider value=".5"></frc-number-slider>

        <frc-number-slider 
          value="25"
          min="0"
          max="100"
          block-increment="25"
        ></frc-number-slider>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-number key="/slider1/value" value="0"></nt-number>
        <nt-number key="/slider1/min" value="-5"></nt-number>
        <nt-number key="/slider1/max" value="5"></nt-number>
        <nt-number key="/slider1/blockIncrement" value="1"></nt-number>

        <frc-number-slider source-key="/slider1"></frc-number-slider>

        <nt-number key="/num1" value="-1"></nt-number>

        <frc-number-slider source-key="/num1"></frc-number-slider>

With attributes and NetworkTables


        <!-- Code for "With attributes and NetworkTables" example -->
        <nt-number key="/num2" value="3"></nt-number>

        <frc-number-slider source-key="/num2" max="5"></frc-number-slider>

        <nt-number key="/slider2/min" value="-20"></nt-number>
        <nt-number key="/slider2/max" value="20"></nt-number>

        <frc-number-slider 
          source-key="/slider2"
          value="6" 
          block-increment="2"
        ></frc-number-slider>
