Examples for Combo Box Chooser

With attributes


        <!-- Code for "With attributes" example -->
        <frc-combobox-chooser 
          name="Combo Box"
          selected="Option 2"
          default="Option 1"
          options='["Option 1", "Option 2", "Option 3", "Option 4"]'
        ></frc-combobox-chooser>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-string 
          key="/SmartDashboard/Autonomous Mode/.name" 
          value="Autonomous Mode"
        ></nt-string>

        <nt-string 
          key="/SmartDashboard/Autonomous Mode/selected" 
          value="Drive Backwards"
        ></nt-string>

        <nt-string 
          key="/SmartDashboard/Autonomous Mode/default" 
          value="None"
        ></nt-string>

        <nt-string-array
          key="/SmartDashboard/Autonomous Mode/options" 
          value='["Drive Backwards", "Drive Forward", "Feature Example", "None"]'
        ></nt-string-array>

        <frc-combobox-chooser 
          source-key="/SmartDashboard/Autonomous Mode"
        ></frc-combobox-chooser>
