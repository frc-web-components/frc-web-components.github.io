Examples for Number Bar

With attributes


        <!-- Code for "With attributes" example -->
        <frc-number-bar value=".5"></frc-number-bar>

        <frc-number-bar 
          value="2"
          min="-10"
          max="20"
          center="5"
          hide-text
          num-tick-marks="3"
        ></frc-number-bar>

        <frc-number-bar 
          value=".7"
          min="0"
          max="1"
          center="0"
        ></frc-number-bar>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-number key="/bar1/value" value="7"></nt-number>
        <nt-number key="/bar1/min" value="0"></nt-number>
        <nt-number key="/bar1/max" value="10"></nt-number>
        <nt-number key="/bar1/center" value="8"></nt-number>
        <nt-boolean key="/bar1/hideText" value="false"></nt-boolean>
        <nt-number key="/bar1/numTickMarks" value="7"></nt-number>

        <frc-number-bar source-key="/bar1"></frc-number-bar>

        <nt-number key="/num1" value="-.8"></nt-number>

        <frc-number-bar source-key="/num1"></frc-number-bar>

With attributes and NetworkTables


        <!-- Code for "With attributes and NetworkTables" example -->
        <nt-number key="/num2" value="3"></nt-number>

        <frc-number-bar source-key="/num2" max="5" hide-text></frc-number-bar>

        <nt-number key="/bar2/min" value="-20"></nt-number>
        <nt-number key="/bar2/max" value="20"></nt-number>

        <frc-number-bar 
          source-key="/bar2"
          value="6" 
          num-tick-marks="2"
        ></frc-number-bar>
