Examples for Accelerometer

With attributes


        <!-- Code for "With attributes" example -->
        <frc-accelerometer value=".5"></frc-accelerometer>

        <frc-accelerometer value=".9" min="0"></frc-accelerometer>

        <frc-accelerometer 
          value="2"
          min="-5"
          max="5"
          precision="4"
          num-tick-marks="3"
        ></frc-accelerometer>

        <frc-accelerometer 
          value="-.5"
          hide-text
          num-tick-marks="10"
        ></frc-accelerometer>

With NetworkTables


        <!-- Code for "With NetworkTables" example -->
        <nt-number key="/accel1/value" value="7"></nt-number>
        <nt-number key="/accel1/min" value="0"></nt-number>
        <nt-number key="/accel1/max" value="10"></nt-number>
        <nt-number key="/accel1/precision" value="8"></nt-number>
        <nt-boolean key="/accel1/hideText" value="false"></nt-boolean>
        <nt-number key="/accel1/numTickMarks" value="7"></nt-number>

        <frc-accelerometer source-key="/accel1"></frc-accelerometer>

        <nt-number key="/num1" value="-.8"></nt-number>

        <frc-accelerometer source-key="/num1"></frc-accelerometer>

        <nt-number key="/LiveWindow/Ungrouped/Accelerometer[0]/Value" value=".5"></nt-number>

        <frc-accelerometer source-key="/LiveWindow/Ungrouped/Accelerometer[0]"></frc-accelerometer>

With attributes and NetworkTables


        <!-- Code for "With attributes and NetworkTables" example -->
        <nt-number key="/num2" value="3"></nt-number>

        <frc-accelerometer source-key="/num2" max="5" hide-text></frc-accelerometer>

        <nt-number key="/accel2/min" value="-20"></nt-number>
        <nt-number key="/accel2/max" value="20"></nt-number>

        <frc-accelerometer 
          source-key="/accel2"
          value="6" 
          num-tick-marks="2"
        ></frc-accelerometer>
